# installation samba 4.7.0, ( PS : c'est l'horreur )

apt install wget make build-essential libacl1-dev libattr1-dev libblkid-dev libgnutls28-dev libreadline-dev pkg-config libsystemd-dev

# installation python2 ( c'est long ) pour pouvoir make le samba 4.7.0 car il l'utilise 
wget https://www.python.org/ftp/python/2.7.18/Python-2.7.18.tgz
tar xzf Python-2.7.18.tgz
cd Python-2.7.18
./configure --enable-optimizations
make -j$(nproc)
make altinstall

# lien python2.7 vers python pour samba
sudo ln -s /usr/local/bin/python2.7 /usr/bin/python

#puis normalement 
python2 ./buildtools/bin/waf configure 
python2 ./buildtools/bin/waf build
python2 ./buildtools/bin/waf install
